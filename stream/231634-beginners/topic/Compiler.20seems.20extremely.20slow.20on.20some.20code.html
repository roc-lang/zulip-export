sequence = \taskList ->
    Task.loop (taskList, List.withCapacity (List.len taskList)) \(tasks, values) ->
        when tasks is
            [task, .. as rest] ->
                value = task!
                Task.ok (Step (rest, List.append values value))

            [] ->
                Task.ok (Done values)
