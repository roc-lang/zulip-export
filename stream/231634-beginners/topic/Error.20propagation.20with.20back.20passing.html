Something like this?

errorPropagation =
   a <- Num.divChecked 2 0 |> Result.next
   b = a - 1
   c <- Num.divChecked 3 b |> Result.next

errorPropagation =
   a <- Num.divChecked 2 2 |> Result.next
   b = a - 1
   c <- Num.divChecked 3 b |> Result.next

errorPropagation =
