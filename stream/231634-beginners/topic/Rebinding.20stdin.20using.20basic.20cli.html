Brendan Hansknecht (Nov 19 2023 at 23:42):

Just tested with this

#include <stdio.h>
#include <stdbool.h>

int main(void){
    size_t no = 0;
    bool line_top = true;

    while (1) {
        int ch = fgetc(stdin);
        if(ch == EOF) break;

        if(line_top)
            printf("%zu ", ++no);

        line_top = (ch == '\n');

        fputc(ch, stdout);
    }
    fflush(stdout);

    freopen("newin", "r", stdin);
    int ch = fgetc(stdin);
    if(ch == EOF) {
        printf("reopened file is just at eof");
    } else {
        printf("the reopened file is useful");
    }
    fflush(stdout);
    return 0;
}

$ clang main.c -o main && echo 7 | ./main
1 7
reopened file is just at eof%
