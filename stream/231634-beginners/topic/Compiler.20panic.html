import pf.Stdin

parse_record : Str -> List U64
parse_record = |text|
    text |> Str.split_on(" ") |> List.keep_oks Str.to_u64

parse_matrix : Str -> List List U64
parse_matrix = |text|
    text |> Str.split_on("\n") |> List.map parse_record

main! = |_args|
    text = Stdin.read_to_end!({})? |> Str.from_utf8?
