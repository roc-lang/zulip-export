main : Task {} []
main =
    content <- readFile "day01/sample" |> await
    Stdout.line "I read the file back. Its contents: \"\(content)\""

readFile : Str -> Task Str []
readFile = \filePath ->
    task =
        File.readUtf8 (Path.fromStr filePath)
    attempt task \result ->
        when result is
            Err _ -> Task.succeed "Error"
            Ok content -> Task.succeed content
