find_first : List(v), (v -> Bool) -> [Just(U64, v), Nothing]
find_first = |l, pred| {
    var $found = Nothing
    var $index = 0
    for el in l {
        if pred(el) {
            $found = Just($index, el)
            break
        }
        $index = $index + 1
    }
    $found
}
