Brendan Hansknecht (Jun 02 2024 at 20:41):

Personally, I think I would just go with something simple:

fizzbuzz = \i ->
    when (i % 3, i % 5) is
        (0, 0) -> "FizzBuzz"
        (0, _) -> "Fizz"
        (_, 0) -> "Buzz"
        _ -> Num.toStr i

Brendan Hansknecht (Jun 02 2024 at 20:51):

A more direct translation would be this:

a = ["Fizz", "", ""]
b = ["Buzz", "", "", "", ""]

getWrap = \list, i ->
    when List.get list (i % List.len list) is
        Ok v -> v
        Err _ -> crash "invalid"

fizzbuzz = \i ->
    when Str.concat (getWrap a i) (getWrap b i) is
        "" -> Num.toStr i
        x -> x
