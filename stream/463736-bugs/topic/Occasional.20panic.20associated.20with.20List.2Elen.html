Anton (Mar 20 2026 at 17:55):

Temporary workaround with the for loop version of keep_oks:

keep_oks = |list, fun| {
    var $dest = []
    for elem in list {
        match fun(elem) {
            Ok(ok_val) => {
                $dest = $dest.append(ok_val)
            }
            Err(_) => {
                $dest = $dest
            }
        }
    }
    $dest
}
