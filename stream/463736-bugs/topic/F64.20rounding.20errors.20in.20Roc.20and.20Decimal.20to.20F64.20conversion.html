score : F64, F64 -> U64
score = \x, y ->
    distance = x * x + y * y
        |> Num.sqrt
    rawScore = (-1.0 / 180.0) * distance * distance + (17.0 / 60.0) * distance - 23.0 / 18.0
    # Need rounded score to remove precision errors with F64 data type
    roundedScore = rawScore * 1000000.0
        |> Num.round
        |> Num.toF64
        |> \n -> n / 1000000.0
    when Num.ceiling roundedScore is
        -1 -> 10
        0 -> 5
        1 -> 1
        _ -> 0
