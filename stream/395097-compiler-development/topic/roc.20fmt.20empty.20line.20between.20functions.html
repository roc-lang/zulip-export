        ## expect "".concat("") == ""
        ## ```
        concat : Str, Str -> Str
        ## Determines whether or not the first Str contains the second.

Becomes:

        is_empty : Str -> Bool
        is_empty = |str| Str.count_utf8_bytes(str) == 0
