So here is the first Roc program to compile with 100% PNC usage:

app [main!] {
    pl: platform "../../basic-cli/platform/main.roc",
}

import pl.Stdout
import pl.Stdin
import pl.Stderr

main! = \_args ->
    tick!({})

tick! = \{} ->
    try Stdout.write!("Enter a number: ")
    when Stdin.line!({}) is
        Ok(str) ->
            num =
                Str.toU8(str)
                |> Result.withDefault(255)
                |> addOne()
            try Stdout.line!("Did you write $(Num.toStr(num)) (Y/n)")
            when Stdin.line!({}) is
                Ok "n" -> Ok({})
                Ok _ -> tick!({})
                Err _ -> Err(Done)

        Err(EndOfFile) ->
            try Stderr.line!("EOF")
            Err(Done)

        Err(StdinErr(_)) ->
            try Stderr.line!("Couldn't read from stdin")
            Err(Done)

addOne = \n ->
    if n == 255 then
        255
    else
        n + 1
