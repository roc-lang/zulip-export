# not sure why => is required
part1 : List(Str) => U64
part1 = |lines| {
    var $pos = 50
    var $count = 0
    for line in lines {
        u8_line = line.to_utf8()
        match u8_line {
            ['L', ..a] => {
                amount = i64_from_ascii_digits(a)
                $pos = ($pos - amount).mod_by(100)
                if ($pos == 0) {
                    $count = $count + 1
                }
            }
            ['R', ..a] => {
                amount = i64_from_ascii_digits(a)
                $pos = ($pos + amount).mod_by(100)
                if ($pos == 0) {
                    $count = $count + 1
                }
            }
            _ => dbg "Error"
        }
    }
    $count
}

i64_from_ascii_digits : List(U8) -> I64
i64_from_ascii_digits = |ascii| {
    ascii.map(|b| (b - '0').to_i64()).fold(0, |n,b| n * 10 + b)
}
